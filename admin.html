// Αντικατάστησε τις αντίστοιχες συναρτήσεις στον κώδικα της V13 function analyzeAttendance() { const input = document.getElementById('mass-input').value.toLowerCase(); // 1. UNIQUE NAMES ONLY: Φιλτράρισμα για να μην υπάρχουν διπλά ονόματα στο ίδιο input let names = input.match(/[a-z0-9_]+/g) || []; names = [...new Set(names)]; const listDiv = document.getElementById('detected-list'); listDiv.innerHTML = ""; matchedData = {}; Object.keys(allCPs).forEach(id => { if(allCPs[id].clan === 'history') return; const roster = (allCPs[id].members || "").toLowerCase(); const leader = (allCPs[id].leader || "").toLowerCase(); let count = 0; names.forEach(n => { if(n.length > 2 && (roster.includes(n) || leader === n)) count++; }); if(count > 0) { // 2. 9-MEMBER CAP: Εμφάνιση προειδοποίησης αν βρέθηκαν πάνω από 9 const displayCount = count > 9 ? `9 (Detected ${count} - Capped at 9)` : count; matchedData[id] = count; listDiv.innerHTML += `
${allCPs[id].name}: ${displayCount}
`; } }); } async function logValidatedEvent(bossId, multiplier) { const checkboxes = document.querySelectorAll('.cp-checkbox'); let activityLogged = false; for (const cb of checkboxes) { if (cb.checked) { const id = cb.id.replace('check-', ''); // Εφαρμογή του Cap των 9 ατόμων κατά την εγγραφή στη βάση const rawCount = matchedData[id]; const finalCount = Math.min(rawCount, 9); const pointsToAdd = finalCount * multiplier; await db.ref('cps/' + id).transaction(cp => { if(cp) { if(!cp.bossStats) cp.bossStats = {}; cp.bossStats[bossId] = (cp.bossStats[bossId] || 0) + finalCount; cp.totalPoints = (cp.totalPoints || 0) + pointsToAdd; } return cp; }); activityLogged = true; } } if(activityLogged) { alert("Event Logged! Duplicates removed & 9-member cap applied."); document.getElementById('mass-input').value = ""; document.getElementById('detected-list').innerHTML = ""; } }
